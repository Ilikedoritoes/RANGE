#define _CRT_SECRURE_NO_WARNINGS
#include <stdlib.h>
#include <stdio.h> 


//      index++ -> index+1
//      index+=2   -> index+2

// יש לכתוב תוכנית אשר מדפיסה את המספרים הזוגים בלבד מ0 עד 100.
void main()
{
    int limit;
    int num;
    scanf("%d %d ", num,limit);


    for (num;limit;num++)
    {
        printf(" -> %d", num);

    }


}

// FOR (מה הערך בהתחלה ; מה הגבול של הערך ; בכה להכפיל ולהגדיל אותו)
